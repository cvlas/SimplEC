working(X) iff
	loc(X)=work,
	not takingBreak(X).
	
resting(X) iff
	loc(X)=pub or onHoliday(X) or (loc(X)=work, takingBreak(X)).



happy(X)=true iff
	rich(X)=true or
	resting(X)=true.

sad(X) iff
	poor(X),
	(working(X) or (loc(X)=home, changingDiapers(X))).

friends(X,Y) iff
	(loc(X)=pub,
	loc(Y)=pub),
	not(rich(X) or rich(Y)).

fighting(X,Y) iff
	( (abrupt(X) or abrupt(Y)), close(X,Y) ),
	not(inactive(X) or inactive(Y) ).

		
			
			


initiate moving(P1,P2) if
	start walking(P1),
	walking(P2),		
	close(P1,P2).
											
terminate moving(P1,P2) if
	end walking(P1).

terminate moving(P1,P2) if
	happens exit(P1).







initiate a(X) if
	not happens b(X),
	not c(X).

initiate c(X) if
	happens d(X).
