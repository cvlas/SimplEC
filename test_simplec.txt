holds working(X) iff
	loc(X)=work,
	not takingBreak(X).
	
holds resting(X) iff
	loc(X)=pub or onHoliday(X) or (loc(X)=work, takingBreak(X)).



holds happy(X)=true iff
	rich(X)=true or
	resting(X)=true.

holds sad(X) iff
	poor(X),
	(working(X) or (loc(X)=home, changingDiapers(X))).

holds friends(X,Y) iff
	(loc(X)=pub,
	loc(Y)=pub),
	not(rich(X) or rich(Y)).

holds fighting(X,Y) iff
	( (abrupt(X) or abrupt(Y)), close(X,Y) ),
	not(inactive(X) or inactive(Y) ).

	start walking(P1),
	holds walking(P2),		
	holds close(P1,P2).

	end walking(P1).

terminate moving(P1,P2) if
	happens exit(P1).
